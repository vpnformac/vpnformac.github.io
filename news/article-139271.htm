3. 核心配置艺术

配置文件中几个关键参数需要特别注意：

json "inbounds": {...}, "outbounds": [ { "protocol": "vmess", "settings": { "vnext": [ { "address": "your_server_ip", "port": 443, "users": [ { "id": "your_uuid", "alterId": 64 } ] } ] }, "streamSettings": { "network": "ws", "security": "tls", "wsSettings": { "path": "/your_path" } } } ]
